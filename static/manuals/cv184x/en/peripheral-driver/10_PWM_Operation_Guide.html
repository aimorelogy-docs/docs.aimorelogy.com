#define PWM_PATH "/sys/class/pwm/pwmchip6"

int pwm_write(const char *path, const char *value) {
    int fd = open(path, O_WRONLY);
    if (fd < 0) {
        perror("EnableFileFailure");
        return -1;
    }
    if (write(fd, value, strlen(value)) < 0) {
        perror("WriteFailure");
        close(fd);
        return -1;
    }

}
